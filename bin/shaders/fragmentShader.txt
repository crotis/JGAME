#version 150

//Executes after vertex shader
//Executes for each pixel to decide what colour it should be. Input colour for 
//each run of the fragmentShader is a linearly interpolated colour value of the vertexShader output colour.  

in vec3 colour;
in vec2 pass_textureCoordinates;

out vec4 out_Color;

uniform sampler2D modelTexture;

void main(void){

    //Samples texture given in sampler2D at the texture coordinates given
    //Gets the colour at those coords and returns it so we can output it to the pixel
	out_Color = texture(modelTexture,pass_textureCoordinates);

}
